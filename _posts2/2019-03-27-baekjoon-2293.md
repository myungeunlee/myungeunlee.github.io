## [1] Answer Code (19. 03. 27)

``` java


import java.util.Scanner;

public class Main {

	static int n, k;
	static long[][] dir;

	public static void main(String[] args) {

		Scanner sc = new Scanner(System.in);
		n = sc.nextInt();
		k = sc.nextInt();
		int[] arr = new int[n];
		int[] ans = new int[k+1];

		for(int i=0; i<n; i++) {
			arr[i] = sc.nextInt();
		}

		ans[0] = 1;
		for(int i=0; i<n; i++) {
			for(int j=1; j<=k; j++) {
				if(j-arr[i] >= 0) {
					ans[j] += ans[j-arr[i]];
				}
			}
		}

		System.out.println(ans[k]);

	}

}
```
