import java.awt.Point;
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class Main {

	static int N, L, R, move;
	static int[][] map;
	static int[][] dir = {
		{0,1},
		{1,0},
		{-1,0},
		{0,-1}
	};

	static void bfs() {


		int nextr, nextc, val, personsum;
		Point now;

		while(true) {
			boolean visited[][] = new boolean[N][N];
			personsum = 0;

			boolean flag = false;
			for(int i=0; i<N; i++) {
				for(int j=0; j<N; j++) {

					if(!visited[i][j]) {
						Queue<Point> que = new LinkedList<>();
						que.add(new Point(i, j));
						ArrayList<Point> sumlist = new ArrayList<>();
						sumlist.add(new Point(i, j));
						personsum = map[i][j];
						visited[i][j] = true;

						while(!que.isEmpty()) {
							now = que.poll();

							for(int d=0; d<4; d++) {
								nextr = now.x + dir[d][0];
								nextc = now.y + dir[d][1];

								if(nextr < 0 || nextc < 0 || nextr >= N || nextc >= N)
									continue;

								if(visited[nextr][nextc])
									continue;

								val = Math.abs(map[nextr][nextc] - map[now.x][now.y]);

								if(L <= val && val <= R) {
									visited[nextr][nextc] = true;
									sumlist.add(new Point(nextr, nextc));
									personsum += map[nextr][nextc];
									flag = true;
									que.add(new Point(nextr, nextc));
								}
							}
						}

						int newpersoncnt = 0;
						if(sumlist.size()>1) {

							newpersoncnt = personsum/sumlist.size();

							for(Point p : sumlist) {
								map[p.x][p.y] = newpersoncnt;
							}
							personsum = 0;
						}
					}
				}
			}

			if(flag)
				move++;
			else
				break;

		}

	}

	public static void main(String[] args) throws IOException{
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine());

		N = Integer.parseInt(st.nextToken());
		L = Integer.parseInt(st.nextToken());
		R = Integer.parseInt(st.nextToken());

		map = new int[N][N];

		for(int i=0; i<N; i++) {
			st = new StringTokenizer(br.readLine());
			for(int j=0; j<N; j++) {
				map[i][j] = Integer.parseInt(st.nextToken());
			}
		}

		move = 0;
		bfs();

		System.out.println(move);
	}

}
