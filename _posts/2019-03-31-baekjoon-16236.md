import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.*;

public class Main {
    
	static int N, fishcnt, ans;
	static int[][] map;
	static boolean[][] visited;
	static int[][] dir = {{-1,0},
  {0,-1},
  {0,1},
  {1,0}};
	
	
	public static int bfs(Shark shark) {
		
		Shark now = shark;
		int nextr= shark.r, nextc = shark.c, nexts = shark.s, nexteat =0 , nextm=shark.move, chkmove, sharksize=2;
		
		while(true) {
			
			Queue<Shark> que = new LinkedList<>();
			que.add(new Shark(nextr, nextc, nexts, nextm));
			visited = new boolean[N][N];
			LinkedList<Shark> fish = new LinkedList<>();
			chkmove = -1;
			
			if(fishcnt == 0) {
				return ans;
			}
			
			while(!que.isEmpty()) {
				now = que.poll();
				
				
				if(chkmove == now.move) {
					break;
				}
				
				for(int d=0; d<4; d++) {
					nextr = now.r + dir[d][0];
					nextc = now.c + dir[d][1];
					
					
					if(nextr < 0 || nextc < 0 || nextr >= N || nextc >= N)
						continue;
					
					if(visited[nextr][nextc] || map[nextr][nextc] > sharksize)
						continue;
					
					if(map[nextr][nextc] > 0 && map[nextr][nextc] < sharksize) {
						fish.add(new Shark(nextr, nextc, nextr*N+nextc, now.move+1));
						chkmove = now.move+1;
					}
					
					que.add(new Shark(nextr, nextc, nextr*N+nextc, now.move+1));
					visited[nextr][nextc] = true;
					
				}
				
			}
			
			if(chkmove == -1) {
				return ans;
			}else {
				if(!fish.isEmpty())
					Collections.sort(fish, new SelectShark());
			}
			
			Shark newshark = fish.get(0);
			nextr = newshark.r;
			nextc = newshark.c;
			nexts = newshark.s;
			nextm = 0;
			nexteat++;
			if(nexteat == sharksize) {
				sharksize++;
				nexteat = 0;
			}
			fishcnt--;
			map[nextr][nextc] = 0;
			ans += chkmove;
		}
	}
	
	
	public static void main(String args[]) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;
        
        N = Integer.parseInt(br.readLine());
        map = new int[N][N];
        visited = new boolean[N][N];
        int tmp;
        Shark shark = null;
        fishcnt = 0;
        ans =0;
        for(int i=0; i<N; i++) {
        	st = new StringTokenizer(br.readLine());
        	
        	for(int j=0; j<N; j++) {
        		tmp = Integer.parseInt(st.nextToken());
        		map[i][j] = tmp;
        		if(tmp==9) {
        			shark = new Shark(i, j, 0, 0);
        			map[i][j] = 0;
        		}else if(tmp!=0) {
        			fishcnt++;
        		}
        		
        		
        	}
        }
        
        System.out.println(bfs(shark));
        
    }
	
	static class SelectShark implements Comparator<Shark>{
		public int compare(Shark s1, Shark s2) {
			if(s1.r < s2.r)
				return -1;
			else if(s1.r == s2.r) {
				if( s1.c < s2.c)
					return -1;
				else
					return 1;
			}else
				return 1;
		}
		
	}
	
	static class Shark{
		int r, c, s, move;

		public Shark(int r, int c, int s, int move) {
			this.r = r;
			this.c = c;
			this.s = s;
			this.move = move;
		}
	}
}
